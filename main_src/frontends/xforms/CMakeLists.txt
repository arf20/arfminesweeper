        add_library(arfminesweeper-xforms MODULE "xforms.c")
        target_compile_definitions(arfminesweeper-xforms PUBLIC _MODULE_)
        target_link_libraries(arfminesweeper-xforms forms)

    else()

    endif()

        target_sources(arfminesweeper PRIVATE "xforms.c")
        target_link_libraries(arfminesweeper forms)

        message(STATUS "Building with static xforms")
    else()
        message(STATUS "Not building with static xforms")
    endif()
