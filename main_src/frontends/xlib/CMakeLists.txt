if (X11_FOUND)
    if (MODULE_XLIB)
        add_library(arfminesweeper-xlib MODULE "xlib.c")
        target_compile_definitions(arfminesweeper-xlib PUBLIC _MODULE_)
        target_link_libraries(arfminesweeper-xlib ${X11_LIBRARIES})
        add_custom_target(symlink_arfminesweeper-xlib ALL
            COMMAND ln -fs "${CMAKE_CURRENT_BINARY_DIR}/*.so" ${MODULE_DIR}
            DEPENDS arfminesweeper-xlib module_dir)
        message(STATUS "Building with module xlib")
    else()
        message(STATUS "Not building with module xlib")
    endif()

    if (STATIC_XLIB)
        target_sources(arfminesweeper PRIVATE "xlib.c")
        target_link_libraries(arfminesweeper ${X11_LIBRARIES})
        target_compile_definitions(arfminesweeper PUBLIC FRONTEND_XLIB)
        message(STATUS "Building with static xlib")
    else()
        message(STATUS "Not building with static xlib")
    endif()
else()
    message(STATUS "No Xlib found")
endif()
